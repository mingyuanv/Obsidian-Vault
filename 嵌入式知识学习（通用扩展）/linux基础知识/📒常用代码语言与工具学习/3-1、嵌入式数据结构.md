sqlink list_create()

    {

        return -1;
    }

    if(pos < 0 || pos > L->last + 1)

    L->last++;

}

			r->next=a;

			r=r->next;
			r->next=NULL;
		}

	//连接剩余节点

## 检查循环队列是否为空

//表示队列为空，返回1，否则返回0

		return -1;

#include <stdio.h>
#include "linkqueue.h"
#include "sqstack.h"

//函数 check 检查队列中的元素是否按升序排列。
int check(linkqueue *lq);

int main(char argc, char *argv[])
{

    linkqueue *lq;
    sqstack *s_hour, *s_five, *s_min;
    
    int value;
    int i,min = 0;

    // 创建一个队列并检查创建是否成功
    if ((lq = create_queue()) == NULL) {
        return -1;
    }
    for(i = 1;i<=27;i++){
        queue_push(lq,i);
    }


    // 创建三个栈，并检查创建是否成功
    if ((s_hour = stack_create(11)) == NULL || 
        (s_five = stack_create(11)) == NULL || 
        (s_min = stack_create(4)) == NULL) {
        return -1;
    }

    while(1){
        min++;// 计数器加一

        // 如果队列不为空，则从队列中弹出一个元素
        if(!queue_empty(lq)){
            value = queue_pop(lq);

            // 尝试将该元素压入 s_min 栈
            if(!stack_full(s_min)){
                stack_push(s_min,value);

            }else{// 若 s_min 栈已满，则尝试将其余元素重新加入队列，并尝试其他栈
                while(!stack_empty(s_min)){
                    queue_push(lq,stack_pop(s_min));
                }

                // 尝试将元素压入 s_five 栈
                if(!stack_full(s_five)){
                    stack_push(s_five,value);
                }else{
                    while(!stack_empty(s_five)){
                        queue_push(lq,stack_pop(s_five));
                    }

                    // 尝试将元素压入 s_hour 栈
                    if(!stack_full(s_hour)){
                        stack_push(s_hour,value);
                    }else{
                        // 若 s_hour 栈也满了，尝试将其中的元素重新加入队列
                        while(!stack_empty(s_hour)){
                            queue_push(lq,stack_pop(s_hour));
                        }
                        queue_push(lq,value);
                        
                        // 检查队列中的元素是否按升序排列，若是则跳出循环
                        if(check(lq)==1){
                            break;
                        }

                    }

                }

            }

        }
    }

    printf("total:%d\n", min);  // 打印总共经过了多少次循环

	// 输出队列中的所有元素
    queue_show(lq);

    return 0;
}



//对队列中元素顺序的检查。
//按升序排列返回 1；否则返回 0；若队列指针为 NULL 返回 -1。
int check(linkqueue *lq)
{
    linklist p;

	if (lq == NULL) {
		printf("lq is NULL\n"); 
		return -1;
	}

    p=lq->front->next; // 从第一个数据节点开始

    // 遍历队列，检查元素是否按升序排列
    while(p && p->next){
        if(p->data < p->next->data){
            p=p->next;
        }else{

            return 0;
        }
    }

    return 1;
}
