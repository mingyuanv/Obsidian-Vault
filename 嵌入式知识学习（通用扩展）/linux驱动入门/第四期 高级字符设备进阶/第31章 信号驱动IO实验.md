## 信号驱动IO简介
### 1 、信号驱动IO不需要应用程序查询设备的状态，一旦设备准备就绪，会触发SIGIO信号，进而调用注册的处理函数。

> 仍旧以钓鱼为例。小马同学喜欢吃新鲜的鱼，但是不想自己钓，所以他请了一个助手来帮他钓鱼，他自己去忙其他的事情（进程不阻塞，立即返回）。如果有鱼上钩**助手会帮忙钓上来（将数据拷贝到指定的缓冲区）**，并立即通知小马同学回来把鱼取走（处理数据）

## 应用程序使用信号驱动IO
### 1 、应用程序使用信号驱动IO的步骤

> 步骤1 ：注册信号处理函数 应用程序使用**signal函数来注册SIGIO信号**的信号处理函数。
> 
> 步骤2： 设置能够**接收这个信号的进程**
> 
> 步骤3： 开启信号驱动IO 通常使用**fcntl函数的F_SETFL命令打开FASYNC标志**。


### 2 、fcntl函数 - 操作文件描述符

#### 函数原型：

```c
int fcntl(int fd,int cmd,...);
```


#### 函数参数

> fd: 被操作的文件描述符
> 
> cmd: 操作文件描述符的命令，**cmd参数决定了要如何操作文件描述符fd**
> 
> …: 根据cmd的参数来决定是不是需要使用第三个参数

#### 操作文件描述符的命令(cmd)

| 命令名           | 描述                                  |
| ------------- | ----------------------------------- |
| F\_DUPFD      | 复制文件描述符                             |
| F\_GETFD      | 获取文件描述符标志                           |
| F\_SETFD      | 设置文件描述符标志                           |
| F\_GETFL      | **获取文件状态标志**                        |
| F\_SETFL      | **设置文件状态标志**                        |
| F\_GETLK      | 获取文件锁                               |
| F\_SETLK      | 设置文件锁                               |
| F\_SETLKW     | 类似F\_SETLK，但等待返回                    |
| F\_GETOWN     | 获取当前接收SIGIO和SIGURG信号的进程ID和进程组ID     |
| **F\_SETOWN** | **设置当前接收SIGIO和SIGURG信号的进程ID和进程组ID** |

### 3 、signal(SIGIO,func) - 用于注册 ​​SIGIO 信号处理函数​​

> `SIGIO`：一种异步通知信号，当文件描述符（如设备文件、套接字）的 I/O 状态发生变化（如数据可读/可写）时，内核会向进程发送此信号
> `func`：用户自定义的函数指针，用于处理接收到的 `SIGIO`信号。函数格式需为 `void func(int sig_num)`，其中 `sig_num`是信号编号（此处为 `SIGIO`）

## 驱动程序实现fasync方法
### 1 、 当应用程序开启信号驱动IO时，会触发驱动中的fasync函数。



### 2 、fasync函数
#### 函数原型

```c
int (*fasync)(int fd,struct file *file,int on);
```


#### 函数参数：

> fp: 要操作的fasync_struct
> 
> sig: 发送的信号
> 
> band: 可读的时候设置成POLLIN ，可写的时候设置成POLLOUT



### 3 、fasync方法的实现
- 1 首先在file_operations结构体中实现fasync函数


- 1 fasync函数会调用fasync_helper函数来操作fasync_struct结构体

- 1 当设备准备好的时候，驱动程序需要调用kill_fasync函数通知应用程序，此时应用程序的SIGIO信号处理函数就会被执行。（write函数中实现）


### 4 、fasync_helper函数
```c
int fasync_helper(int fd,struct file *file,int on,struct fasync_struct **fapp);
```


### 5、kill_fasync函数 - 发送指定的信号
```c
void kill_fasync(struct fasync_strct **fp,int sig,int band);
```


> fp: 要操作的fasync_struct
> 
> sig: 发送的信号
> 
> band: **可读的时候设置成POLLIN ，可写的时候设置成POLLOUT**

## 编写应用程序read.c
### 1 、使用信号驱动IO读取数据


### 2 、代码
```c
#include <stdio.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>
#include <stdlib.h>
#include <poll.h>
#include <fcntl.h>
#include <signal.h>

int fd;
char buf1[32]={0};

//SIGIO信号的信号处理函数
void func(int signum){

    read(fd,buf1,sizeof(buf1));
    printf("buf1 is %s \n,",buf1); //打印读取的数据
}




int main(int argc,char *argv[])
{


    int ret;
    int flags;

    fd=open("/dev/test",O_RDWR); //打开/dev/test设备，阻塞式访问
    if(fd<0){
        printf("open error\n");
    }

    //步骤一：使用signal函数注册SIGIO信号的信号处理函数
    signal(SIGIO,func);

    //步骤二：将文件描述符 fd的 SIGIO/SIGURG 信号接收者绑定为当前进程
    //F_SETOWN 设置当前接收的SIGIO的进程ID
    fcntl(fd,F_SETOWN,getpid());
    
    //获取文件描述符标志
    flags=fcntl(fd,F_GETFD);

    //以​信号驱动模式​​监听文件描述符
    //触发内核调用设备驱动的fasync函数
    fcntl(fd,F_SETFL,flags|FASYNC);



    while(1);
    close(fd);

    return 0;
}
```

### 3 、 fcntl(fd,F_SETFL,flags|FASYNC);

> ​**​启用异步通知机制​**​
> `FASYNC`标志告知内核：当前进程希望以​**​信号驱动模式​​监听文件描述符**`fd`的状态变化（如数据可读、可写）。当设备或文件就绪时，内核自动向进程发送`SIGIO`信号，无需进程主动轮询
> ​**​触发驱动层回调​**​
> 设置`FASYNC`标志会**触发内核调用设备驱动的`fasync()`函数**（需驱动实现）。该函数通过`fasync_helper`将进程注册到驱动的异步通知队列中，建立信号发送通道

##  驱动程序编写
### 1 、代码总揽
```c
#include <linux/init.h>
#include <linux/module.h>
#include <linux/kdev_t.h>
#include <linux/fs.h>
#include <linux/cdev.h>
#include <linux/uaccess.h>
#include <linux/io.h>
#include <linux/wait.h>
#include <linux/poll.h>
#include <linux/fcntl.h>
#include <linux/signal.h>

struct device_test{

    dev_t dev_num;//设备号
    int major;  
    int minor;

    struct cdev cdev_test;  // 字符设备结构体
    struct class *class;    // 类指针
    struct device *device;  // 设备指针

    char kbuf[32];  // 内核缓冲区,存放数据
    int flag;  // 标志位，表示是否有数据可读

    struct fasync_struct *fasync;

};

// 实例化一个设备结构体变量
struct device_test dev1;

// 定义并初始化等待队列头，用于阻塞读取
DECLARE_WAIT_QUEUE_HEAD(read_wq);


static int cdev_test_open(struct inode *inode,struct file *file){

    // 设置私有数据为设备结构体地址，方便其他函数使用
    file->private_data=&dev1;

    printk("This is cdev_test_open\r\n");

    return 0;
}

static int cdev_test_release(struct inode *inode, struct file *file)
{
    printk("This is cdev_test_release\r\n");
    return 0;
}


static ssize_t cdev_test_write (struct file *file,const char __user *buf,size_t size,loff_t *off){

    struct device_test *test_dev=(struct device_test *)file->private_data;

    if(copy_from_user(test_dev->kbuf,buf,size)!=0){

        printk("copy_from_user  error\r\n");

        return -1;
    }

    // 设置标志位，表示有新数据到达
    test_dev->flag=1;

    //唤醒 等待队列（read_wq）中处于 可中断睡眠状态（TASK_INTERRUPTIBLE）的进程。
    //通常用于通知等待资源（如数据、硬件状态）的进程：“资源已就绪，请继续执行”。
    wake_up_interruptible(&read_wq);

    return 0;
}


static ssize_t cdev_test_read (struct file *file, char __user *buf,size_t size,loff_t *off){

    struct device_test *test_dev=(struct device_test *)file->private_data;

    // 如果是非阻塞模式并且没有数据，则立即返回
    if(file->f_flags & O_NONBLOCK){
        if(test_dev->flag !=1){

            return -EAGAIN;
        }
    }

    // 阻塞等待直到有数据可用
    wait_event_interruptible(read_wq, test_dev->flag);



    if(copy_to_user(buf,test_dev->kbuf,strlen(test_dev->kbuf))!=0){

        printk("copy_from_user  error\r\n");

        return -1;
    }

    printk("This is cdev_test_read\r\n");

    return 0;
}


static __poll_t cdev_test_poll(struct file *file,struct poll_table_struct *p){

    struct device_test *test_dev=(struct device_test *)file->private_data;  //设置私有数据

    __poll_t mask=0;

    // 将当前进程（调用 poll 或 select 的进程）挂起到 wait_address 指向的等待队列中，以便在等待队列对应的事件发生时被唤醒。
    poll_wait(file,&read_wq,p);//应用阻塞

    // 如果有数据可读，设置 POLLIN 标志
    if(test_dev->flag==1){

        mask |= POLLIN;
    }

    return mask;
}


//fasync方法的实现
static int cdev_test_fasync(int fd,struct file *file,int on){

    struct device_test *test_dev=(struct device_test *)file->private_data;  //设置私有数据

    return fasync_helper(fd,file,on,&test_dev->fasync);
}





struct file_operations cdev_test_fops={

    .owner = THIS_MODULE, //将owner字段指向本模块，可以避免在模块的操作正在被使用时卸载该模块
    .open=cdev_test_open,
    .read=cdev_test_read,
    .write=cdev_test_write,
    .poll=cdev_test_poll,
    .release=cdev_test_release,
    .fasync=cdev_test_fasync

};


static int __init chr_fops_init(void){

    int ret;

    /*1 创建设备号*/
    ret=alloc_chrdev_region(&dev1.dev_num,0,1,"alloc_name");
    if(ret<0){
        goto err_chrdev;
    }
    printk("alloc_chrdev_region is ok\n");

    //打印设备号
    dev1.major=MAJOR(dev1.dev_num);
    dev1.minor=MINOR(dev1.dev_num);
    printk("major is %d \r\n", dev1.major); //打印主设备号
    printk("minor is %d \r\n", dev1.minor); //打印次设备号

    /*2 初始化cdev*/
    dev1.cdev_test.owner=THIS_MODULE;
    cdev_init(&dev1.cdev_test,&cdev_test_fops);

    /*3 添加一个cdev,完成字符设备注册到内核*/
   ret=cdev_add(&dev1.cdev_test,dev1.dev_num,1);
    if(ret<0)
    {
        goto  err_chr_add;
    }

    /*4 创建类*/
    dev1.class=class_create(THIS_MODULE,"test");
    if(IS_ERR(dev1.class)){

        ret=PTR_ERR(dev1.class);
        goto err_class_create;
    }

    /*5  创建设备*/
    dev1.device=device_create(dev1.class,NULL,dev1.dev_num,NULL,"test");
    if(IS_ERR(dev1.device))
    {
        ret=PTR_ERR(dev1.device);
        goto err_device_create;
    }

return 0;

//错误处理
err_device_create:
    class_destroy(dev1.class);

err_class_create:
    cdev_del(&dev1.cdev_test);

err_chr_add:
    unregister_chrdev_region(dev1.dev_num,1);

err_chrdev:
    return ret;

}



static void __exit chr_fops_exit(void) //驱动出口函数
{
    /*注销字符设备*/
    unregister_chrdev_region(dev1.dev_num, 1); //注销设备号
    cdev_del(&dev1.cdev_test);                 //删除cdev
    device_destroy(dev1.class, dev1.dev_num);       //删除设备
    class_destroy(dev1.class);                 //删除类
}


module_init(chr_fops_init);
module_exit(chr_fops_exit);

MODULE_LICENSE("GPL v2");
MODULE_AUTHOR("topeet");

```
